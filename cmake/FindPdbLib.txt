# This file defines:
# - PDBLIB_LIBRARIES
# - PDBLIB_INCLUDE_DIRS
# - PDBLIB_FOUND
#

FIND_LIBRARY(PDBLIB_LIBRARIES
  NAMES
    PdbLib
  PATHS
    /usr/lib
    /usr/local/lib
    /opt/local/lib
    /home/tusi/works/pdbtm_4.0/PdbLib/lib	
)

FIND_PATH(PDBLIB_INCLUDE pdblib.h
    /usr/include
    /usr/local/include/pdblib
    /home/tusi/works/pdbtm_4.0/PdbLib/include
)

SET(PDBLIB_FOUND FALSE)

IF(PDBLIB_LIBRARIES AND PDBLIB_INCLUDE)
  SET(PDBLIB_FOUND TRUE)
  SET(PDBLIB_INCLUDE_DIRS ${PDBLIB_INCLUDE})
  SET(PDBLIB_LIBRARY_DIRS ${PDBLIB_LIBRARIES})
  MESSAGE ( STATUS "Found the PDBLIB headers at ${PDBLIB_INCLUDE}" )
  MESSAGE ( STATUS "Found the PDBLIB libraries at ${PDBLIB_LIBRARIES}" )
ELSE()
  MESSAGE ( FATAL "The PDBLIB module not found" )
  MESSAGE ( STATUS "Found the PDBLIB headers at ${PDBLIB_INCLUDE}" )
  MESSAGE ( STATUS "Found the PDBLIB libraries at ${PDBLIB_LIBRARIES}" )
ENDIF(PDBLIB_LIBRARIES AND PDBLIB_INCLUDE)
